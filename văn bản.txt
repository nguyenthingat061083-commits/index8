<script>
    // Hàm chuyển đổi Tab mượt mà giữa các chức năng
    function switchTab(evt, tabId) {
        // Tắt tất cả panel cũ
        const panels = document.querySelectorAll('.tab-panel');
        panels.forEach(panel => panel.classList.remove('active'));

        // Gỡ trạng thái active của tất cả các nút
        const buttons = document.querySelectorAll('.tab-btn');
        buttons.forEach(btn => btn.classList.remove('active'));

        // Kích hoạt panel và nút được chọn
        document.getElementById(tabId).classList.add('active');
        evt.currentTarget.classList.add('active');
    }

    // Đổi Màu Accent (Các thành phần highlight, checkbox, line phân tách, text cam)
    function changeAccentColor(color) {
        document.documentElement.style.setProperty('--accent-color', color);
        
        // Tự động tính toán sắc độ tối hơn cho hover hiệu ứng nút bấm
        let r = parseInt(color.slice(1, 3), 16);
        let g = parseInt(color.slice(3, 5), 16);
        let b = parseInt(color.slice(5, 7), 16);
        r = Math.max(0, Math.floor(r * 0.85)).toString(16).padStart(2, '0');
        g = Math.max(0, Math.floor(g * 0.85)).toString(16).padStart(2, '0');
        b = Math.max(0, Math.floor(b * 0.85)).toString(16).padStart(2, '0');
        document.documentElement.style.setProperty('--accent-hover', `#${r}${g}${b}`);
    }

    // Đổi Màu Nền (Màu chủ đạo chính của Menu Panel và thanh Sidebar)
    function changeMenuBgColor(color) {
        document.documentElement.style.setProperty('--bg-content', color);
        
        // Tự động tính toán để Sidebar có màu sẫm hơn màu nền Content chính 25% giúp phân định rõ ràng
        let r = parseInt(color.slice(1, 3), 16);
        let g = parseInt(color.slice(3, 5), 16);
        let b = parseInt(color.slice(5, 7), 16);
        
        let sideR = Math.max(0, Math.floor(r * 0.70)).toString(16).padStart(2, '0');
        let sideG = Math.max(0, Math.floor(g * 0.70)).toString(16).padStart(2, '0');
        let sideB = Math.max(0, Math.floor(b * 0.70)).toString(16).padStart(2, '0');
        
        let itemR = Math.min(255, Math.floor(r * 1.30)).toString(16).padStart(2, '0');
        let itemG = Math.min(255, Math.floor(g * 1.30)).toString(16).padStart(2, '0');
        let itemB = Math.min(255, Math.floor(b * 1.30)).toString(16).padStart(2, '0');

        document.documentElement.style.setProperty('--bg-sidebar', `#${sideR}${sideG}${sideB}`);
        document.documentElement.style.setProperty('--bg-item', `#${itemR}${itemG}${itemB}`);
    }

    // Logic Kéo Thả Menu chuyên nghiệp (Cầm giữ Sidebar để kéo)
    const menu = document.getElementById("mod-menu");
    const dragHandle = document.getElementById("sidebar-drag");
    let isDragging = false;
    let offsetX, offsetY;

    // Căn giữa menu lần đầu tiên tải trang
    menu.style.left = (window.innerWidth - 680) / 2 + "px";
    menu.style.top = (window.innerHeight - 440) / 2 + "px";

    dragHandle.addEventListener("mousedown", (e) => {
        // Chỉ cho phép kéo khi không bấm trúng nút chuyển tab
        if (e.target.closest('.tab-btn')) return;
        
        isDragging = true;
        offsetX = e.clientX - menu.offsetLeft;
        offsetY = e.clientY - menu.offsetTop;
        document.addEventListener("mousemove", dragMove);
        document.addEventListener("mouseup", dragStop);
    });

    function dragMove(e) {
        if (!isDragging) return;
        menu.style.left = (e.clientX - offsetX) + "px";
        menu.style.top = (e.clientY - offsetY) + "px";
    }

    function dragStop() {
        isDragging = false;
        document.removeEventListener("mousemove", dragMove);
        document.removeEventListener("mouseup", dragStop);
    }
</script>

</body>
</html>
